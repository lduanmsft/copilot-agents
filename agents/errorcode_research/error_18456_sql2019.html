// Converts internal login errors to external errors on a case-by-case basis
// By default returns GATEWAY_DATABASE_UNAVAILABLE error unless
// error is explicitly handled
long ConvertInternalError(__in long lError, __inout_opt ELoginFailedState& lfState)
{
    switch (lError)
    {
    case EX_NUMBER(GATEWAY2, GATEWAY_GEODR_OFFLINE_SECONDARY_ACCESS_DENY):
    case EX_NUMBER(GATEWAY3, GATEWAY_AKV_INACCESSIBLE):
        return lError;

    case EX_NUMBER(GATEWAY2, GATEWAY_EXTERNAL_CONNECTION_DENIED):
        lfState = x_elfDenyExternalConnectionsOn;
        return LOGON_FAILED;

    default:
        return EX_NUMBER(GATEWAY, GATEWAY_DATABASE_UNAVAILABLE);
    }
}
